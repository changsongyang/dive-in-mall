```shell
# 全局安装
npm install -g eslint
# 本地安装
npm install eslint --save-dev
npm install babel-eslint --save-dev
# 生成配置文件
eslint --init
# 运行
eslint yourfile.js
# 自动修复
eslint yourfile.js --fix
```
